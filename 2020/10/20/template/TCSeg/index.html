长链剖分

简介

长链剖分也是一种树剖，和轻重链剖分相比，它唯一的区别就是确定重儿子的方式：长链剖分是以能到达深度最深的儿子作为重儿子的，其他都与轻重链剖分大同小异。

应用

O(n)​ ~ O(n log n) 求 K 级祖先

首先利用倍增求出每个点的 2i2^i2i 级祖先，然后对于每条链，在顶点处记录顶点向上的 aaa 个祖先和向下的 aaa 个链上的儿子 ( aaa 为当前链长度 )，并且预处理 [1, n][1,\ n][1, n] 中每一个数二进制的最高位，记作 hhh。

在查询的时候，先跳到当前点的 2hk2^{h_k}2hk​ 级祖先，那么剩下的级数 k′<2hkk' \lt 2^{h_k}k′<2hk​，所以当前跳到的节点的长链长度 lenlenlen 满足 len≥2hk>k′len \ge 2^{h_k} \gt k'len≥2hk​>k′，那么我们可以再跳到当前跳到节点所在链的顶部，用之前预处理出来的祖先儿子数组获取答案。

实现：( 代码有删减 )

void PreDFS(int now){
	maxD(now) = dep(now) = dep(fa[now][0]) + 1;

	for(int i = hd[now]; i; i = e[i].next){

		for(int j = 0; j <= 19; j++){
			fa[e[i].v][j+1] = fa[fa[e[i].v][j]][j];
		}

		PreDFS(e[i].v);
		if(maxD(e[i].v) > maxD(hSon(now))){
			hSon(now) = e[i].v;
			maxD(now) = maxD(e[i].v);
		}
	}
}

void SegDFS(int now, int top){
	top(now) = top; len(now) = maxD(now) - dep(top(now)) + 1;

	if(now == top){
		for(int i = 0, o = now; i <= maxD(now) - dep(now); i++){
			U[now].push_back(o); o = fa[o][0];
		}
		for(int i = 0, o = now; i <= maxD(now) - dep(now); i++){
			V[now].push_back(o); o = hSon(o);
		}
	}

	if(hSon(now)){
		SegDFS(hSon(now), top);

		for(int i = hd[now]; i; i = e[i].next){
			if(e[i].v == fa[now][0] || e[i].v == hSon(now)){
				continue;
			}
			SegDFS(e[i].v, e[i].v);
		}
	}
}

int GetAnc(int now, int k){
	if(!k){
		return now;
	}
	now = fa[now][g[k]], k -= ((1 << g[k]) + (dep(now) - dep(top(now)))), now = top(now);
	return k >= 0 ? U[now][k] : V[now][-k];
}

优化树形动归

其实这个思想类似于 Dsu on treeDsu\ on\ treeDsu on tree，就是在一类状态设计中有深度这一维度的树形动归中，我们可以通过指针操作做到 O(1)O(1)O(1) 复制重儿子节点的信息，然后达到优化掉一个 nnn 的时间以及空间复杂度的效果。

实现：( 代码有缩减 )

int *f[MAXN];
int mem[MAXN], *ptr = mem;
// mem就是用来分配内存的，没别的用途
// *ptr 记录内存大小

void PreDFS(int now, int fa){
	for(int i = hd[now]; i; i = e[i].next){
		if(e[i].v == fa){
			continue;
		}
		PreDFS(e[i].v, now);
		if(dep(e[i].v) > dep(hSon(now))){
			hSon(now) = e[i].v;
		}
	}
	dep(now) = dep(hSon(now)) + 1;
}

void AnsDFS(int now, int fa){
	f[now][0] = 1;
	if(hSon(now)){
		f[hSon(now)] = f[now] + 1;
		// 共享内存, O(1) 传递重儿子答案
		AnsDFS(hSon(now), now);
		// ……
	}
	for(int i = hd[now]; i; i = e[i].next){
		if(e[i].v == fa || e[i].v == hSon(now)){
			continue;
		}
		f[e[i].v] = ptr; ptr += dep(e[i].v);
		// 给轻儿子分配内存
		AnsDFS(e[i].v, now);
		for(int j = 1; j <= dep(e[i].v); j++){
			f[now][j] += f[e[i].v][j-1]; // 暴力复制轻儿子信息
            // DP 转移
		}
	}
}
