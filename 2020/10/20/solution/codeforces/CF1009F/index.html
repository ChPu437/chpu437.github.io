CF1009F Dominant Indices 题解

题面：CodeForces 洛谷

长链剖分优化树形 DPDPDP 练手题。
这道题的状态设计和转移方程很好得出，如下：

设 fi, jf_{i,\ j}fi, j​ 表示当前节点为 iii，距离当前节点深度为 jjj 的节点数，则有：

fi, j=∑fu, j−1,u∈i 的儿子f_{i,\ j} = \sum f_{u,\ j-1},\quad u \in i\ 的儿子fi, j​=∑fu, j−1​,u∈i 的儿子

但是朴素的转移空间和时间都是 O(n2)O(n^2)O(n2) 的，无法接受。

观察到转移方程的其中一维是深度，可以用长链剖分进行优化[1]，将深度一维优化掉，便可以在时间和空间均为 O(n)O(n)O(n) 的复杂度内解决问题。

代码：

# include <iostream>
# include <cstdio>
# define MAXN 1000005

struct edge{
	int v, next;
}e[MAXN<<1]; int hd[MAXN], cntE;
struct node{
	int hSon, dep;
# define hSon(x) nd[x].hSon
# define dep(x) nd[x].dep
}nd[MAXN];
int *f[MAXN];
int mem[MAXN], *ptr = mem;
// mem就是用来分配内存的，没别的用途
// *ptr 记录内存大小
int ans[MAXN];

void AddE(int u, int v){
	e[++cntE] = (edge){v, hd[u]};
	hd[u] = cntE;
}

void PreDFS(int now, int fa){
	for(int i = hd[now]; i; i = e[i].next){
		if(e[i].v == fa){
			continue;
		}
		PreDFS(e[i].v, now);
		if(dep(e[i].v) > dep(hSon(now))){
			hSon(now) = e[i].v;
		}
	}
	dep(now) = dep(hSon(now)) + 1;
}

void AnsDFS(int now, int fa){
	f[now][0] = 1;
	if(hSon(now)){
		f[hSon(now)] = f[now] + 1;
		// 共享内存, O(1) 传递重儿子答案
		AnsDFS(hSon(now), now);
		ans[now] = ans[hSon(now)] + 1;
	}
	for(int i = hd[now]; i; i = e[i].next){
		if(e[i].v == fa || e[i].v == hSon(now)){
			continue;
		}
		f[e[i].v] = ptr; ptr += dep(e[i].v);
		// 给轻儿子分配内存
		AnsDFS(e[i].v, now);
		for(int j = 1; j <= dep(e[i].v); j++){
			f[now][j] += f[e[i].v][j-1];
			if(f[now][j] > f[now][ans[now]] || (f[now][j] == f[now][ans[now]] && j < ans[now])){ // DP 转移
				ans[now] = j;
			}
		}
	}
	if(f[now][ans[now]] == 1){
		ans[now] = 0;
	}
}

int main(){
	int n;

	scanf("%d", &n);
	for(int i = 1, x, y; i <= n-1; i++){
		scanf("%d%d", &x, &y);
		AddE(x, y); AddE(y, x);
	}

	PreDFS(1, 0);
	f[1] = ptr; ptr += dep(1);
	// 为根节点分配内存
	AnsDFS(1, 0);

	for(int i = 1; i <= n; i++){
		printf("%d\n", ans[i]);
	}

	return 0;
}
