本文最后更新于：2020年10月8日 凌晨

题面及题解

T1 NYG的背包

题解

考虑贪心。

首先不难发现一定要先处理加入后可以使体积增大的，这样可以尽量让后面有足够的空间。所以分为加入后体积增大和减小的分类讨论。

对于所有增大的，由于增大总量不变，而且加入元素后体积只增不减，所以应该按照原体积 (aaa) 从小到大排序，然后依次加入。

对于所有减小的，考虑先放能让体积增大量( bbb )大的，这样才能保证后面的元素有足够大的空间放置。

在模拟放置过程中，由于已经是最优方案，一旦体积不足，直接跳出即可。

复杂度 O(nlog⁡n+n)O(n\log n + n)O(nlogn+n)。

注意本题需要开 long long！

代码

// 肯定是先放能增大体积的再放不能的
// 分为增大体积和减小体积的考虑
// 增大体积的先后顺序不会影响增大的值，所以先放小的
// 增大体积放进去只增不降，不会出现能放下大体积却放不下小体积的情况，所以某个体积放不下一定是无解的
// 减小体积的放进去只降不增
// 准确来说不能叫减小体积，而是放下这个物品后背包又增大了一些
// 应该考虑让这个增大的值足够大，这样才能给后面的物品留够空间
// 而这样放无法放下的物品其他方法也放不下 ? 

# include <iostream>
# include <cstdio>
# include <algorithm>
# define int long long
# define pii std::pair<int, int>
# define mkp std::make_pair
# define MAXN 100010

inline void read(int & x){
	x = 0; int ch = getchar();
	for(	;!isdigit(ch); ch = getchar());
	for(	; isdigit(ch); ch = getchar()){
		x = x * 10 + ch - '0';
	}
}

pii pls[MAXN], mns[MAXN];
int cntP, cntM;

bool CmpP(pii x, pii y){
	return x.first == y.first ? x.second > y.second : x.first < y.first;
}
bool CmpM(pii x, pii y){
	return x.second > y.second;
}

int main(){
	int t;
	read(t);

	for(int n, h; t; t--){
		cntP = cntM = 0;
		read(n), read(h);
		for(int i = 1, x, y; i <= n; i++){
			read(x), read(y);
			if(y - x >= 0){
				pls[++cntP] = mkp(x, y);
			}
			else{
				mns[++cntM] = mkp(x, y);
			}
		}

		std::sort(pls+1, pls+cntP+1, CmpP);
		std::sort(mns+1, mns+cntM+1, CmpM);

		bool flag = 1; 

		for(int i = 1; i <= cntP && flag; i++){
			if(h < pls[i].first){
				flag = 0; break;
			}
			else{
				h += (pls[i].second - pls[i].first);
			}
		}
        
		for(int i = 1; i <= cntM && flag; i++){
			if(h < mns[i].first){
				flag = 0; break;
			}
			else{
				h -= (mns[i].first - mns[i].second);
			}
		}

		if(flag){
			printf("Yes\n");
		}
		else{
			printf("No\n");
		}
	}

	return 0;
}

T2 NYG的动态数点

题解

经过一通观察，不难发现性质：一个区间合法，当且仅当区间最小值与区间 gcd⁡\gcdgcd 相等。

所以我们可以用 STSTST 表预处理区间最小值与 gcd⁡\gcdgcd ，然后二分区间长度进行二分答案即可。

复杂度 O(nlog⁡n)O(n \log n)O(nlogn)。

代码

// 二分区间长度，检查这个长度下有没有区间 GCD 和 最小值相等的

# include <iostream>
# include <cstdio>
# include <cmath>
# define MAXN 500005

template<typename T>inline void read(T & x){
	x = 0; int ch = getchar();
	for(	;!isdigit(ch); ch = getchar());
	for(	; isdigit(ch); ch = getchar()){
		x = x * 10 + ch - '0';
	}
}

unsigned a[MAXN], gcd[MAXN][25], min[MAXN][25]; int n;
unsigned p[MAXN];

unsigned GCD(unsigned x, unsigned y){
	return y ? GCD(y, x%y) : x;
}

bool Check(int len){
	int q = log2(len), k = n + 1 - p[q], j;
	len--;

	for(int i = 1; i <= k; i++){
		j = i + len;
		if(std::min(min[i][q], min[j - p[q] + 1][q]) == GCD(gcd[i][q], gcd[j-p[q]+1][q])){
			return true;
		}
	}

	return false;
}

int ansP[MAXN], cntA; 

int main(){
	read<int>(n);
	int m = log2(n);
// ST 表预处理区间最小值和 GCD
	for(int i = 1; i <= n; i++){
		read<unsigned>(a[i]);
		gcd[i][0] = min[i][0] = a[i];
	}
	for(int i = 0; i <= m; i++){
		p[i] = (1 << i);
	}
	for(int j = 1, k; j <= m; j++){
		k = n + 1 - p[j];
		for(int i = 1; i <= k; i++){
			min[i][j] = std::min(min[i][j-1], min[i+p[j-1]][j-1]);
			gcd[i][j] = GCD(gcd[i][j-1], gcd[i+p[j-1]][j-1]);
		}
	}
// 二分答案
	int l = 1, r = n;
	while(l <= r){
		int mid = (l + r) >> 1;
		if(Check(mid)){
			l = mid + 1;
		}
		else{
			r = mid - 1;
		}
	}

	if(r == 1){
		printf("%d 0\n", n);
		for(int i = 1; i < n; i++){
			printf("%d\n", n);
		}
	}
	else{
		int q = log2(r), k = n + 1 - p[q], j;
		r--;
		for(int i = 1; i <= k; i++){
			j = i + r;
			if(std::min(min[i][q], min[j-p[q]+1][q]) == GCD(gcd[i][q], gcd[j-p[q]+1][q])){
				ansP[++cntA] = i;
			}
		}
		printf("%d %d\n", cntA, r);
		for(int i = 1; i <= cntA; i++){
			printf("%d ", ansP[i]);
		}
	}

	return 0;
}

T3 金币

题目描述

LSLSLS 得到了一张藏宝图，他准备去探宝，地图由 nnn 行 mmm 列的点构成，每一个点有 qqq 个金币，没有金币的店为空地，LSLSLS 可以选择在任意一个有金币的点出发，这一个点横纵向延伸所在块的所有金币全部拿到，但是 LSLSLS 一天只能进行一次探险，所以他想知道他需要多少天才能拿到所有的金币以及每天最多能拿到多少金币。

输入格式

第 111 行，两个整数 mmm，nnn，第 222 至 n+1n+1n+1 行，每行 mmm 个整数，代表每个点的金币数。

输出格式

第 111 行，探险的天数 kkk，

第 222 行，kkk 个整数，代表每天最多能拿到多少金币。

数据范围

对于 30%30\%30% 的数据，n≤10n\le10n≤10，m≤10m \le 10m≤10，

对于 50%50\%50% 的数据，n≤1000n \le 1000n≤1000，m≤1000m \le 1000m≤1000，

对于 70%70\%70% 的数据，n≤3000n \le 3000n≤3000，m≤3000m \le 3000m≤3000，块的总数 ≤10\le 10≤10，

对于 100%100\%100% 的数据，n≤6000n \le 6000n≤6000，m≤6000m \le 6000m≤6000，块的总数 ≤10000\le 10000≤10000，每个点的金币数 ≤9\le 9≤9。

题解

这题主要是题面比较难读，但是读懂了会发现它异常的水……

就是查找连通块，直接暴力枚举点跑 BFSBFSBFS 即可。

注意读入的 nnn 和 mmm 是反的！

代码

// 一次只能拿同一块的
// 大力 BFS 即可
// 统计块数和每块大小

# include <iostream>
# include <cstdio>
# include <algorithm>
# include <queue>
# define MAXN 6010
# define pii std::pair<int, int>
# define mkp std::make_pair

inline void read(int & x){
	x = 0; int ch = getchar();
	for(	;!isdigit(ch); ch = getchar());
	for(	; isdigit(ch); ch = getchar()){
		x = x * 10 + ch - '0';
	}
}

const int toX[] = {0, 1, 0,-1, 0};
const int toY[] = {0, 0,-1, 0, 1};

int gra[MAXN][MAXN];
char ln[MAXN];
int cost[1000005], cntC;

void BFS(pii from){
	cntC++;
	std::queue<pii>Q;

	Q.push(from);
	cost[cntC] += gra[from.first][from.second];
	gra[from.first][from.second] = 0;

	while(Q.size()){
		pii now = Q.front(); Q.pop();
		for(int i = 1, nxtX, nxtY; i <= 4; i++){
			nxtX = now.first + toX[i], nxtY = now.second + toY[i];
			if(gra[nxtX][nxtY]){
				cost[cntC] += gra[nxtX][nxtY];
				gra[nxtX][nxtY] = 0;
				Q.push(mkp(nxtX, nxtY));
			}
		}
	}
}

bool CmpC(int x, int y){
	return x > y;
}

int main(){
	int n, m;

	read(m), read(n);

	for(int i = 1; i <= n; i++){
		scanf("%s", ln+1);
		for(int j = 1; j <= m; j++){
			gra[i][j] = ln[j] - '0';
		}
	}

	for(int i = 1; i <= n; i++){ // 
		for(int j = 1; j <= m ;j++){
			if(gra[i][j]){
				BFS(mkp(i, j));
			}
		}
	}

	std::sort(cost+1, cost+cntC+1, CmpC);

	printf("%d\n", cntC);

	for(int i = 1; i <= cntC; i++){
		printf("%d ", cost[i]);
	}

	return 0;
}

T4 沙漠绿化

题目描述

我们定义沙漠是满足如下性质的图：

没有重边和自环，且图中任意一条边至多属于一个简单环。

给定一片沙漠，从中删去任意多条边，使得删去边之后的图是一个森林，求有多少种方法，答案对 998244353998244353998244353 取模。

输入格式

第一行两个整数 NNN，MMM 表示图中的点数和边数。

接下来 MMM 行，每行两个整数 uuu，vvv，表示一条边的两个端点。

输出格式

输出一行一个整数表示答案。

数据范围

对于 20%20\%20% 的数据，图中只存在至多一个环，

对于 50%50\%50% 的数据，满足任意一个点至多属于一个简单环，

对于 100%100\%100% 的数据，1≤N≤3×1051 \le N \le 3 \times 10^51≤N≤3×105，1≤M≤5×1051 ≤ M ≤ 5 \times 10^51≤M≤5×105。

题解

对于链上的边很显然取不取都行，方案为 2n2^n2n 种，

对于环上的点有一种方案就是都取的方案是不合法的，总方案为 2n−12^n-12n−1 种。

直接跑 DFSDFSDFS 找环即可。

代码

# include <iostream>
# include <cstdio>
# define MAXN 300005
# define MAXM 500005

const long long MOD = 998244353;

struct edge{
	int v, next;
}e[MAXM<<1]; int hd[MAXN], cntE;

void AddE(int u, int v){
	e[++cntE] = (edge){v, hd[u]};
	hd[u] = cntE;
}

int siz[MAXN], cntCyc; // 环的大小，环的个数
int dep[MAXN];
long long two[MAXM];

void DFS(int now, int fa){
	dep[now] = dep[fa] + 1;
	for(int i = hd[now]; i; i = e[i].next){
		if(e[i].v == fa){
			continue;
		}
		if(!dep[e[i].v]){
			DFS(e[i].v, now);
		}
		else{
			if(dep[e[i].v] < dep[now]){
				siz[++cntCyc] = dep[now] - dep[e[i].v] + 1;
			}
		}
	}
}

int main(){
	int n, m;

	scanf("%d%d", &n, &m);

	two[0] = 1;
	for(int i = 1; i <= 500000; i++){
		two[i] = (two[i-1] * 2ll) % MOD;
	}

	for(int i = 1, u, v; i <= m; i++){
		scanf("%d%d", &u, &v);
		AddE(u, v); AddE(v, u);
	}

	for(int i = 1; i <= n; i++){
		if(!dep[i]){
			DFS(i, 0);
		}
	}

	long long ans = 1;

	for(int i = 1; i <= cntCyc; i++){
		ans = (ans * (two[siz[i]]-1+MOD)) % MOD;
		m -= siz[i];
	}

	// printf("%d\n", m);

	printf("%lld", (ans * two[m]) % MOD);

	return 0;
}
