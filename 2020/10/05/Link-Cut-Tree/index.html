[模板][数据结构] Link Cut Tree

先粘个板子，晚上补解释

# include <iostream>
# include <cstdio>
# define MAXN 100005

struct stack{
	int stk[MAXN], topS = 0;
	void push(int x){
		stk[++topS] = x;
	}
	void pop(){
		if(topS){
			topS--;
		}
	}
	int top(){
		return stk[topS];
	}
	void clear(){
		topS = 0;
	}
	int size(){
		return topS;
	}
}S;
struct LCT{ 
	int fa, son[2];
	int val, sum;
	bool tagR;
# define fa(x) tr[x].fa
# define lSon(x) tr[x].son[0]
# define rSon(x) tr[x].son[1]
# define val(x) tr[x].val
# define sum(x) tr[x].sum
}tr[MAXN];

void Rotate(int x); // 注意这个 Rotate 和普通 Splay 的 Rotate 不一样
void Splay(int x); // 注意这个 Splay 和普通 Splay 的 Splay 不一样
void PushUp(int now);
void PushDown(int now);
void Reverse(int now);
bool IsRoot(int now);
void Access(int x);
void MakeRoot(int now);
int FindRoot(int now);
void Split(int x, int y);
void Link(int x, int y);
void Cut(int x, int y);

int main(){
	int n, m;

	scanf("%d%d", &n, &m);

	for(int i = 1; i <= n; i++){
		scanf("%d", &val(i));
	}

	for(int opt, x, y; m; m--){
		scanf("%d%d%d", &opt, &x, &y);
		if(opt == 0){
			Split(x, y);
			printf("%d\n", sum(y));
		}
		else if(opt == 1){
			Link(x, y);
		}
		else if(opt == 2){
			Cut(x, y);
		}
		else if(opt == 3){
			Splay(x), val(x) = y;
		}
	}

	return 0;
}

void Cut(int x, int y){
	MakeRoot(x);
	if(FindRoot(y) == x && fa(y) == x && (!lSon(y))){
		fa(y) = rSon(x) = 0;
		PushUp(x);
	}
}

void Link(int x, int y){
	MakeRoot(x);
	if(FindRoot(y) != x){
		fa(x) = y;
	}
}

void Split(int x, int y){
	MakeRoot(x); Access(y); Splay(y);
}

int FindRoot(int now){
	Access(now); Splay(now);
	while(lSon(now)){
		PushDown(now);
		now = lSon(now);
	}
	Splay(now);
	return now;
}

void MakeRoot(int now){
	Access(now); Splay(now);
	Reverse(now); tr[now].tagR ^= 1;
}

void Access(int x){
	int y = 0;
	while(x){
		Splay(x);
		rSon(x) = y;
		PushUp(x);
		y = x, x = fa(x);
	}
}

bool IsRoot(int now){ // 利用 LCT 认父不认子的性质
	return !((lSon(fa(now)) == now) || (rSon(fa(now)) == now));
}

void Reverse(int now){
	std::swap(lSon(now), rSon(now));
}

void PushDown(int now){
	if(tr[now].tagR){
		if(lSon(now)){
			Reverse(lSon(now));
			tr[lSon(now)].tagR ^= 1;
		}
		if(rSon(now)){
			Reverse(rSon(now));
			tr[rSon(now)].tagR ^= 1;
		}
		tr[now].tagR = 0;
	}
}

void PushUp(int now){
	sum(now) = sum(lSon(now)) ^ sum(rSon(now)) ^ val(now);
}

void Splay(int x){ // 则直接 Splay 到树根
	int y = x, z;
	S.clear(); S.push(y);
	while(!IsRoot(y)){
		y = fa(y);
		S.push(y);
	}
	while(S.size()){
		PushDown(S.top());
		S.pop();
	}
	while(!IsRoot(x)){
		y = fa(x); z = fa(y);
		if(!IsRoot(y)){
			((lSon(y) == x) ^ (lSon(z) == y)) ? Rotate(x) : Rotate(y);
		}
		Rotate(x);
		// 这两句核心代码还是一样的
	}
	PushUp(x);
}

void Rotate(int x){
	int y = fa(x), z = fa(y); bool k = (rSon(y) == x); int w = tr[x].son[k^1];
	if(!IsRoot(y)){ // 这里是和普通 Splay 不同的
		tr[z].son[rSon(z) == y] = x;
	}
	tr[y].son[k] = w, tr[x].son[k^1] = y;
	if(w){
		fa(w) = y;
	}
	fa(y) = x, fa(x) = z;

	PushUp(y);
}
